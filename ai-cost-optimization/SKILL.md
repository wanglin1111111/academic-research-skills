### 缓存策略设计

```python
# 智能缓存策略
class SmartCache:
    def __init__(self, redis_client, ttl=3600):
        self.redis = redis_client
        self.ttl = ttl
    
    def get_cache_key(self, query, params):
        """生成缓存键"""
        import hashlib
        content = f"{query}:{str(sorted(params.items()))}"
        return f"cache:{hashlib.md5(content.encode()).hexdigest()}"
    
    def get(self, query, params):
        """获取缓存"""
        key = self.get_cache_key(query, params)
        result = self.redis.get(key)
        if result:
            return json.loads(result)
        return None
    
    def set(self, query, params, result):
        """设置缓存"""
        key = self.get_cache_key(query, params)
        self.redis.setex(key, self.ttl, json.dumps(result))
    
    def should_cache(self, query_type, user_tier):
        """判断是否应该缓存"""
        # 高频查询必缓存
        if query_type in ["common", "faq"]:
            return True
        # 付费用户延长缓存时间
        if user_tier == "premium":
            return True
        return False
```
