    def calculate_time_savings(self, current_workflow, optimized_workflow):
        """计算时间节省"""
        # 基于行业基准数据估算
        benchmarks = {
            'coding': {'ai_assisted': 0.4, 'traditional': 1.0},
            'design': {'ai_assisted': 0.3, 'traditional': 1.0},
            'writing': {'ai_assisted': 0.5, 'traditional': 1.0},
            'analysis': {'ai_assisted': 0.6, 'traditional': 1.0}
        }
        
        total_savings = 0
        for task in current_workflow:
            task_type = task['type']
            if task_type in benchmarks:
                savings = (benchmarks[task_type]['traditional'] -

                total_savings += savings * task['duration']
        
        return total_savings

### 创业机会评估模型

```python
# 创业机会评估
class OpportunityEvaluator:
    def __init__(self):
        self.criteria = {
            'market': {'weight': 0.25, 'factors': ['size', 'growth', 'competition']},
            'feasibility': {'weight': 0.25, 'factors': ['tech', 'team', 'capital']},
            'timing': {'weight': 0.2, 'factors': ['trend', 'window', 'readiness']},
            'value': {'weight': 0.2, 'factors': ['pain', 'willingness_to_pay', 'retention']},
            'differentiation': {'weight': 0.1, 'factors': ['uniqueness', 'defensibility']}
        }
    
    def evaluate_opportunity(self, opportunity):
        """评估创业机会"""
        scores = {}
        for criterion, config in self.criteria.items():
            criterion_score = self.evaluate_criterion(opportunity, criterion)
            scores[criterion] = {
                'score': criterion_score,
                'weighted': criterion_score * config['weight']
            }
        
        total_score = sum(s['weighted'] for s in scores.values())
        
        return {
            'total_score': total_score,
            'should_pursue': total_score >= 70,
            'breakdown': scores,
            'recommendations': self.generate_recommendations(scores)
        }
```
